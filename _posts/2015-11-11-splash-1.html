plotly.js

plotly.js is an open source, interactive, declarative graphing library built on top of d3.js and stack.gl.

Plotly graphs are interactive. Click-and-drag to zoom, double-click to autoscale, shift-click to pan, select legend entries to toggle. 
Plotly.d3.csv('https://raw.githubusercontent.com/plotly/datasets/master/gapminderDataFiveYear.csv', function(rows){
  var traces = [];
  ['Asia', 'Europe', 'Africa', 'Oceania', 'Americas'].forEach(function(continent) {
    var x = rows.map(function(row){   if(row.continent==continent && +row.year === 2007) return row.lifeExp;});
    var y = rows.map(function(row){   if(row.continent==continent && +row.year === 2007) return row.gdpPercap;});
    var text = rows.map(function(row){if(row.continent==continent && +row.year === 2007) return row.country;});
    var size = rows.map(function(row){if(row.continent===continent && +row.year === 2007) return row.pop;}),
    var sizeRef = Math.max.apply(null, rows.map(function(row){return row.pop;})) / (2*3.14*1000);

    traces.push({
      x: x, y: y, text: text
      name: continent, mode: "markers",
      marker: {size: size, sizemode: "area", sizeref: sizeref}
    });
  });

  var layout = {
    xaxis: {title: "Life Expectancy"}, yaxis: {title: "GDP per Capita", type: "log"},
    margin: {t: 20}, hovermode: "closest"
  };

  Plotly.plot(document.getElementById('my-graph'), traces, layout)
});
plotly.js charts are described declaritively as JSON objects. Every aspect of the charts (the colors, the grids, the data) has a corresponding JSON attribute. Here, we remove the x-axis grid lines with xaxis: {showgrid: false}, the width of the lines with {line: {width: 1}}. View all of the possible attributes. 
Plotly.d3.csv('https://raw.githubusercontent.com/plotly/datasets/master/wind_speed_laurel_nebraska.csv', function(rows){
  var trace = {
    x: rows.map(function(row){return row['Time']; }),
    y: rows.map(function(row){return row['10 Min Sampled Avg']; }),
    line: {
      width: 1
    },
    error_y: {
      array: rows.map(function(row){return row['10 Min Std Dev']; }),
      thickness: 0.5,
      width: 0
    }
  };

  var layout = {
    yaxis: {title: "Wind Speed"},
    xaxis: {showgrid: false},
    margin: {l: 40, b: 40, t: 20}
  };

  Plotly.plot(document.getElementById('wind-speed'), [trace], layout);
});

plotly.js provides abstractions to the types of statistical and scientific charts that you would find in packages like matplotlib, ggplot2, or MATLAB. 
Plotly.d3.json('https://plot.ly/~DanielCarrera/13.json', function(figure){
  var trace = {
    x: figure.data[0].x, y: figure.data[0].y, z: figure.data[0].z,
    type: 'contour', autocolorscale: false,
    colorscale: [[0,"rgb(  0,  0,  0)"],[0.3,"rgb(230,  0,  0)"],[0.6,"rgb(255,210,  0)"],[1,"rgb(255,255,255)"]],
    reversescale: true, zmax: 2.5, zmin: -2.5
  };
  var layout = {
    title: 'turbulence simulation',
    xaxis: {title: 'radial direction', showline: true, mirror: 'allticks', ticks: 'inside'},
    yaxis: {title: 'vertical direction', showline: true, mirror: 'allticks', ticks: 'inside'},
    margin: {l: 40, b: 40, t: 60},
    annotations: [{
      showarrow: false,
      text: 'Credit: Daniel Carrera',
      x: 0, y: 0, xref: 'paper', yref: 'paper'
    }]
  }
  Plotly.plot(document.getElementById('contour-plot'), [trace], layout);
});
stack.gl extends the possibilities of charting in the web browser, allowing interactive 3D graphing or 2D scatter plots, exceeding the inherit performance limitations of SVG in the browser.
By abstracting charts to a declaritive JSON structure, plotly.js is used a browser-based charting library in Python, R, and MATLAB.
